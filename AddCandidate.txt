API endpoint name: AddCandidate.php

POST parameters:
	postAuthKey: our auth key
	name: String
	electionId: Integer
	constituencyName: String
	partyName: String
	img: String(may be blank) path of candidate image
	boothId: String id of booth

Response parameters:
	success: boolean
	validAuth: boolean
	validBooth: boolean(false for invalid or logged out booth)
	validParty: boolean
	validElection: boolean
	validConstituency: boolean
	validCandidate: boolean
	candidateId: Integer Id of candidate (-1 on failure)
