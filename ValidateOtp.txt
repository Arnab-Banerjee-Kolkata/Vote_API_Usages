API endpoint name: ValidateOtp.php


POST parameters:
        postAuthKey: our auth key
        electionId: Integer id of election
        aadhaarNo: String aadhaar number
        category: String category(booth or voter)
        boothId: String id of booth
        otp: Integer OTP
        
        
Response parameters:
        success: boolean
        validAuth: boolean
        validCategory: boolean
       
        For validating voter OTP from remote booth    
            validBooth: boolean(false for invalid booth or logged out booth)
            validStatus: boolean(false if voter has already voted)
            validOtp: boolean(false for wrong otp)
            boothOtp: Integer OTP of booth(-1 in case of invalid voter OTP)
    
    
    
        For validating booth OTP from polling device
            validBooth: boolean(false for invalid booth or logged out booth)
            validApproval: boolean(false if voter is not approved to vote)
            validOtp: boolean(false for wrong otp)
